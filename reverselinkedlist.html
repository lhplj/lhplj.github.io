public class ReverseLinkedList {

    class Node<T> {
        private T value;
        private Node<T> next;
    }

    public static void main(String[] args) {
        ReverseLinkedList.Node<Integer> head = build();
        print(head);
        print(reverse(head));
    }

    private static Node<Integer> build() {
        ReverseLinkedList x = new ReverseLinkedList();
        Node<Integer> head = x.new Node<>();
        head.value = 0;
        Node<Integer> cur = head;
        for (int i = 1; i < 10; i++) {
            Node<Integer> next = x.new Node<>();
            next.value = i;
            cur.next = next;

            cur = next;
        }
        return head;
    }

    /**
     * 打印链表
     * @param head
     */
    private static void print(Node<Integer> head) {
        if (head == null) {
            return;
        }
        List<Integer> list = new ArrayList<>();
        list.add(head.value);
        Node<Integer> next = head.next;
        while (next != null) {
            list.add(next.value);
            next = next.next;
        }
        System.out.println(JSON.toJSONString(list));
    }

    /**
     * 反转后返回新head
     * @param head
     * @return
     */
    private static Node<Integer> reverse(Node<Integer> head) {
        Node<Integer> pre = null;
        Node<Integer> node = head;
        Node<Integer> next = head.next;
        while (next != null) {
            next = node.next;
            node.next = pre;

            pre = node;
            if (next == null) {
                break;
            }
            node = next;
        }
        return node;
    }

}
